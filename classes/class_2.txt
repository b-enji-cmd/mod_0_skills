class Freezer
 

 Attributes


 brand(string)
 temperature_gauge(int)
 freezer_contents(hash) |key = food_name(string) , value = quantity (int) |
 is_door_open(boolean)
 freezer_light_on (boolean)
 number_of_shelves (int)
 



 Methods

 open_door(sets is_door_open && is_freezer_light_on to true)
 add_frozen_food(adds a new key value pair to hash)
 set_temperature(changes temperature_gauge to given integer)
 increase_capacity(increments number_of_shelves) 
 break_shelf(decreases number_of_shelves by 1, decreases value of the last key value pair by 3)
